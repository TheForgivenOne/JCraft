# Convex Setup with Next.js Monorepo

## Environment Variables

Create a `.env.local` file in your project root with the following variables:

```env
# Convex Configuration
NEXT_PUBLIC_CONVEX_URL=https://your-convex-url.convex.cloud
CONVEX_DEPLOYMENT=your-deployment-name
CONVEX_ADMIN_KEY=your-admin-key
```

The `NEXT_PUBLIC_CONVEX_URL` is the most important variable and is automatically generated when you set up your Convex deployment.

## Client Configuration

### Convex Client Provider

Create a client provider component to wrap your application:

```tsx
// apps/web/components/ConvexClientProvider.tsx
'use client';

import { ReactNode } from 'react';
import { ConvexReactClient } from 'convex/react';
import { ConvexProviderWithClerk } from 'convex/react-clerk';
import { useAuth } from '@clerk/nextjs';

if (!process.env.NEXT_PUBLIC_CONVEX_URL) {
  throw new Error('Missing NEXT_PUBLIC_CONVEX_URL in your .env file');
}

const convex = new ConvexReactClient(process.env.NEXT_PUBLIC_CONVEX_URL);

export default function ConvexClientProvider({
  children,
}: {
  children: ReactNode;
}) {
  return (
    <ConvexProviderWithClerk client={convex} useAuth={useAuth}>
      {children}
    </ConvexProviderWithClerk>
  );
}
```

### Root Layout Configuration

In your Next.js root layout, ensure ClerkProvider wraps ConvexClientProvider:

```tsx
// apps/web/app/layout.tsx
import type { Metadata } from 'next';
import './globals.css';
import { ClerkProvider } from '@clerk/nextjs';
import ConvexClientProvider from '@/components/ConvexClientProvider';

export const metadata: Metadata = {
  title: 'Your App',
  description: 'Generated by create next app',
};

export default function RootLayout({
  children,
}: Readonly<{
  children: React.ReactNode;
}>) {
  return (
    <html lang="en">
      <body>
        <ClerkProvider>
          <ConvexClientProvider>{children}</ConvexClientProvider>
        </ClerkProvider>
      </body>
    </html>
  );
}
```

## Schema Definition

Define your database schema in `convex/schema.ts`:

```ts
import { defineSchema, defineTable } from 'convex/server';
import { v } from 'convex/values';

export default defineSchema({
  users: defineTable({
    name: v.string(),
    email: v.string(),
    clerkId: v.string(),
    createdAt: v.timestamp(),
  }).index('by_clerk_id', ['clerkId']),

  projects: defineTable({
    name: v.string(),
    description: v.optional(v.string()),
    ownerId: v.id('users'),
    createdAt: v.timestamp(),
  }).index('by_owner', ['ownerId']),

  tasks: defineTable({
    title: v.string(),
    description: v.optional(v.string()),
    projectId: v.id('projects'),
    completed: v.boolean(),
    createdAt: v.timestamp(),
    updatedAt: v.timestamp(),
  }).index('by_project', ['projectId']),
});
```

## Query and Mutation Functions

Create query and mutation functions in the `convex/` directory:

```ts
// convex/tasks.ts
import { query, mutation } from './_generated/server';
import { v } from 'convex/values';
import { Id } from './_generated/dataModel';

// Query to get all tasks for a project
export const getProjectTasks = query({
  args: {
    projectId: v.id('projects'),
  },
  handler: async (ctx, args) => {
    return await ctx.db
      .query('tasks')
      .withIndex('by_project', q => q.eq('projectId', args.projectId))
      .collect();
  },
});

// Mutation to create a new task
export const createTask = mutation({
  args: {
    title: v.string(),
    description: v.optional(v.string()),
    projectId: v.id('projects'),
    completed: v.boolean(),
  },
  handler: async (ctx, args) => {
    return await ctx.db.insert('tasks', {
      title: args.title,
      description: args.description,
      projectId: args.projectId,
      completed: args.completed,
      createdAt: Date.now(),
      updatedAt: Date.now(),
    });
  },
});
```

## Using Convex in Components

Use Convex queries and mutations in your React components:

```tsx
// Example component using Convex
'use client';

import { useQuery, useMutation } from 'convex/react';
import { api } from '../../convex/_generated/api';

export default function TaskList({ projectId }: { projectId: string }) {
  const tasks = useQuery(api.tasks.getProjectTasks, { projectId });
  const createTask = useMutation(api.tasks.createTask);

  if (!tasks) return <div>Loading...</div>;

  return (
    <div>
      <h2>Tasks</h2>
      <ul>
        {tasks.map(task => (
          <li key={task._id}>{task.title}</li>
        ))}
      </ul>
      <button
        onClick={() => {
          createTask({
            title: 'New Task',
            description: 'Task description',
            projectId: projectId as any,
            completed: false,
          });
        }}
      >
        Add Task
      </button>
    </div>
  );
}
```

## Authentication with Clerk

To store user data when they sign in with Clerk:

```ts
// Hook to store user data in Convex when authenticated with Clerk
import { useUser } from '@clerk/clerk-react';
import { useConvexAuth } from 'convex/react';
import { useEffect, useState } from 'react';
import { useMutation } from 'convex/react';
import { api } from '../convex/_generated/api';
import { Id } from '../convex/_generated/dataModel';

export function useStoreUserEffect() {
  const { isLoading, isAuthenticated } = useConvexAuth();
  const { user } = useUser();
  const [userId, setUserId] = useState<Id<'users'> | null>(null);
  const storeUser = useMutation(api.users.store);

  useEffect(() => {
    if (!isAuthenticated) {
      return;
    }

    async function createUser() {
      const id = await storeUser();
      setUserId(id);
    }
    createUser();
    return () => setUserId(null);
  }, [isAuthenticated, storeUser, user?.id]);

  return {
    isLoading: isLoading || (isAuthenticated && userId === null),
    isAuthenticated: isAuthenticated && userId !== null,
  };
}
```

## Best Practices for Monorepos

1. **File Structure**: Keep your Convex functions in the root `convex/` directory, not within individual app directories
2. **Environment Variables**: Use workspace-level environment variables that can be accessed by your Next.js app
3. **Dependencies**: Install Convex as a dependency in your web app package
4. **Authentication Flow**: Ensure ClerkProvider wraps ConvexClientProvider in your layout
5. **Indexing**: Use indexes on frequently queried fields to optimize performance
6. **Type Safety**: Leverage Convex's automatic type generation for end-to-end type safety
